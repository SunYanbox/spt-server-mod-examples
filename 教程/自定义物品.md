# 自定义物品

你可以通过克隆现有物品来创建新的自定义物品，甚至创建会随机开出物品的战利品箱。

**源文件：** 
- [18CustomItemService/](../18CustomItemService/) - 自定义物品
- [18.1CustomItemServiceLootBox/](../18.1CustomItemServiceLootBox/) - 战利品箱

## 一、创建自定义物品

通过 `CustomItemService` 克隆现有物品并修改属性。

### 基本步骤

```csharp
using SPTarkov.Server.Core.Services;

public class MyMod(
    CustomItemService customItemService
) : IOnLoad
{
    public Task OnLoad()
    {
        var newItem = new NewItemFromCloneDetails
        {
            // 要克隆的源物品
            ItemTplToClone = ItemTpl.SHOTGUN_MP18_762X54R_SINGLESHOT_RIFLE,
            
            // 新物品 ID（必须是有效的 MongoDB ObjectId）
            NewId = "677eed5f2e040616bc7246b6",
            
            // 物品分类树中的位置
            ParentId = "5447b6094bdc2dc3278b4567",
            
            // 跳蚤市场价格
            FleaPriceRoubles = 50000,
            
            // 商店价格
            HandbookPriceRoubles = 42500,
            
            // 商店显示分类
            HandbookParentId = "5b5f78e986f77447ed5636b1",
            
            // 本地化名称
            Locales = new Dictionary<string, LocaleDetails>
            {
                { "en", new LocaleDetails
                    {
                        Name = "MP-18 12g",
                        ShortName = "Custom MP18",
                        Description = "A custom MP18 chambered in 12G"
                    }
                },
                { "zh", new LocaleDetails
                    {
                        Name = "MP-18 12号",
                        ShortName = "定制 MP18",
                        Description = "一把定制版的 MP18，发射12号口径弹药"
                    }
                }
            }
        };
        
        customItemService.CreateItemFromClone(newItem);
        return Task.CompletedTask;
    }
}
```

### 覆盖物品属性

```csharp
var newItem = new NewItemFromCloneDetails
{
    // ... 其他配置
    
    OverrideProperties = new TemplateItemProperties
    {
        // 修改弹药类型
        Chambers = new List<Slot>
        {
            new Slot
            {
                Name = "Caliber",
                Props = new SlotProperties
                {
                    Filters = new List<Filter>
                    {
                        new Filter
                        {
                            Filter = new List<MongoId>
                            {
                                ItemTpl.AMMO_12G_70MM_BUCKSHOT,
                                ItemTpl.AMMO_12G_70MM_SLUG
                            }
                        }
                    }
                }
            }
        }
    }
};
```

## 二、创建战利品箱

战利品箱是可以打开获得随机物品的特殊容器。

### 创建战利品箱物品

```csharp
using SPTarkov.Server.Core.Models.Spt.Config;

public class LootBoxMod(
    CustomItemService customItemService,
    DatabaseServer databaseServer,
    ConfigServer configServer
) : IOnLoad
{
    private readonly InventoryConfig _inventoryConfig = configServer.GetConfig<InventoryConfig>();
    private Dictionary<MongoId, TemplateItem>? _itemDb;
    
    private const string CrateId = "my_custom_crate";

    public Task OnLoad()
    {
        _itemDb = databaseServer.GetTables().Templates.Items;
        
        // 克隆密封武器箱
        var crateItem = new NewItemFromCloneDetails
        {
            ItemTplToClone = "6489b2b131a2135f0d7d0fcb",  // 密封武器箱
            ParentId = "62f109593b54472778797866",
            NewId = CrateId,
            FleaPriceRoubles = 50000,
            HandbookPriceRoubles = 40000,
            HandbookParentId = "5b5f78e986f77447ed5636b1",
            Locales = new Dictionary<string, LocaleDetails>
            {
                { "en", new LocaleDetails
                    {
                        Name = "My Loot Crate",
                        ShortName = "Crate",
                        Description = "A mysterious crate with random items"
                    }
                }
            }
        };
        
        customItemService.CreateItemFromClone(crateItem);
        
        // 关键：修改物品名称使其脱离原有逻辑
        var customItem = _itemDb.GetValueOrDefault(CrateId);
        customItem.Name = CrateId;  // 必须修改！
        
        // 配置战利品池
        _inventoryConfig.RandomLootContainers[CrateId] = new RewardDetails
        {
            RewardCount = 6,      // 开出6个物品
            FoundInRaid = true,   // 标记为战局中获得
            RewardTplPool = new Dictionary<MongoId, double>
            {
                // 物品ID: 权重（权重越高越容易被选中）
                { ItemTpl.DRINK_PACK_OF_MILK, 1 },
                { ItemTpl.FOOD_CAN_OF_TUSHONKA, 1 },
                { ItemTpl.MEDKIT_IFAK, 0.5 },  // 权重较低
                { ItemTpl.BARTER_PHYSICAL_BITCOIN, 0.1 }  // 稀有物品
            }
        };
        
        return Task.CompletedTask;
    }
}
```

## 完整示例

```csharp
using SPTarkov.DI.Annotations;
using SPTarkov.Server.Core.DI;
using SPTarkov.Server.Core.Models.Common;
using SPTarkov.Server.Core.Models.Spt.Config;
using SPTarkov.Server.Core.Models.Spt.Mod;
using SPTarkov.Server.Core.Models.Utils;
using SPTarkov.Server.Core.Servers;
using SPTarkov.Server.Core.Services;

namespace MyMod;

public record ModMetadata : AbstractModMetadata
{
    public override string ModGuid { get; init; } = "com.example.customitem";
    public override string Name { get; init; } = "自定义物品示例";
    public override string Author { get; init; } = "开发者";
    public override SemanticVersioning.Version Version { get; init; } = new("1.0.0");
    public override SemanticVersioning.Range SptVersion { get; init; } = new("~4.0.0");
    public override string License { get; init; } = "MIT";
}

[Injectable(TypePriority = OnLoadOrder.PostDBModLoader + 1)]
public class CustomItemExample(
    ISptLogger<CustomItemExample> logger,
    CustomItemService customItemService,
    DatabaseServer databaseServer,
    ConfigServer configServer
) : IOnLoad
{
    private readonly InventoryConfig _inventoryConfig = configServer.GetConfig<InventoryConfig>();
    private Dictionary<MongoId, TemplateItem>? _itemDb;
    
    private const string CustomCrateId = "677eed5f2e040616bc7246b6";

    public Task OnLoad()
    {
        _itemDb = databaseServer.GetTables().Templates.Items;
        
        // 创建战利品箱
        CreateLootCrate();
        
        logger.Success("自定义物品创建完成！");
        return Task.CompletedTask;
    }
    
    private void CreateLootCrate()
    {
        // 克隆密封武器箱
        var crateItem = new NewItemFromCloneDetails
        {
            ItemTplToClone = "6489b2b131a2135f0d7d0fcb",
            ParentId = "62f109593b54472778797866",
            NewId = CustomCrateId,
            FleaPriceRoubles = 50000,
            HandbookPriceRoubles = 40000,
            HandbookParentId = "5b5f78e986f77447ed5636b1",
            Locales = new Dictionary<string, LocaleDetails>
            {
                { "en", new LocaleDetails
                    {
                        Name = "Mystery Box",
                        ShortName = "Mystery",
                        Description = "What's inside? Open to find out!"
                    }
                },
                { "zh", new LocaleDetails
                    {
                        Name = "神秘箱子",
                        ShortName = "神秘箱",
                        Description = "里面是什么？打开看看！"
                    }
                }
            }
        };
        
        customItemService.CreateItemFromClone(crateItem);
        
        // 修改名称以脱离原有逻辑
        var item = _itemDb.GetValueOrDefault(CustomCrateId);
        item.Name = CustomCrateId;
        
        // 配置战利品池
        _inventoryConfig.RandomLootContainers[CustomCrateId] = new RewardDetails
        {
            RewardCount = 3,
            FoundInRaid = true,
            RewardTplPool = new Dictionary<MongoId, double>
            {
                { ItemTpl.DRINK_PACK_OF_MILK, 2 },
                { ItemTpl.MEDKIT_SALEWA_FIRST_AID_KIT, 1 },
                { ItemTpl.BARTER_PHYSICAL_BITCOIN, 0.1 }
            }
        };
    }
}
```

## 小贴士

1. **NewId 必须是有效的 MongoDB ObjectId**：可以在线生成
2. **战利品箱要修改 Name**：否则会被空投等系统使用，导致不可预期的行为
3. **权重控制概率**：权重越高，物品越容易被选中
4. **ParentId 决定分类**：选择正确的父分类让物品在正确的地方显示
5. **添加多语言**：支持中文和英文让更多玩家使用
