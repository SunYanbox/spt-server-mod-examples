# 命令系统

你可以扩展 SPT 的命令系统，添加新的聊天命令。

**源文件：** 
- [21CustomCommandoCommand/](../21CustomCommandoCommand/) - Commando 命令
- [22CustomSptCommand/](../22CustomSptCommand/) - SPT 命令

## 两种命令类型

| 类型 | 前缀 | 示例 |
|------|------|------|
| Commando 命令 | 自定义 | `test talk` |
| SPT 命令 | `spt` | `spt getName xxx` |

## 一、Commando 命令

创建自定义前缀的命令。

### 基本结构

```csharp
using SPTarkov.DI.Annotations;
using SPTarkov.Server.Core.Models.Common;
using SPTarkov.Server.Core.Models.Eft.Dialog;
using SPTarkov.Server.Core.Services;

[Injectable]
public class MyCommandoCommand(
    DatabaseServer databaseServer,
    MailSendService mailSendService
) : IChatCommand
{
    // 命令前缀
    public string CommandPrefix => "test";
    
    // 支持的子命令列表
    public List<string> Commands => ["talk", "info"];

    public string GetCommandHelp(string command)
    {
        return command switch
        {
            "talk" => "用法: test talk - 测试对话",
            "info" => "用法: test info - 显示信息",
            _ => null
        };
    }

    public ValueTask<string> Handle(string command, UserDialogInfo commandHandler, 
        MongoId sessionId, SendMessageRequest request)
    {
        switch (command)
        {
            case "talk":
                mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, 
                    "你好！这是一个测试消息！");
                break;
                
            case "info":
                var globals = databaseServer.GetTables().Globals;
                mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, 
                    $"行走速度: {globals.Configuration.WalkSpeed.X}");
                break;
        }
        
        return ValueTask.FromResult(request.DialogId);
    }
}
```

**使用方式：** 在聊天框输入 `test talk` 或 `test info`

## 二、SPT 命令

扩展内置的 `spt` 命令。

### 基本结构

```csharp
using SPTarkov.DI.Annotations;
using SPTarkov.Server.Core.Models.Common;
using SPTarkov.Server.Core.Models.Eft.Dialog;
using SPTarkov.Server.Core.Services;
using SPTarkov.Server.Core.Helpers;

[Injectable]
public class MySptCommand(
    MailSendService mailSendService,
    ItemHelper itemHelper
) : ISptCommand
{
    // 子命令名称
    public string Command => "itemName";
    
    // 帮助文本
    public string CommandHelp => "用法: spt itemName <物品ID> - 查询物品名称";

    public ValueTask<string> PerformAction(UserDialogInfo commandHandler, 
        MongoId sessionId, SendMessageRequest request)
    {
        // 解析命令
        // 格式: spt itemName 5449016a4bdc2d6f028b456f
        var parts = request.Text.Split(" ");
        
        if (parts.Length < 3)
        {
            mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, 
                "请提供物品ID，例如: spt itemName 5449016a4bdc2d6f028b456f");
            return ValueTask.FromResult(request.DialogId);
        }
        
        var itemId = parts[2];
        var item = itemHelper.GetItem(itemId);
        
        if (item.Key)
        {
            mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, 
                $"物品名称: {item.Value?.Properties?.Name ?? "未知"}");
        }
        else
        {
            mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, 
                $"未找到物品: {itemId}");
        }
        
        return ValueTask.FromResult(request.DialogId);
    }
}
```

**使用方式：** 在聊天框输入 `spt itemName 5449016a4bdc2d6f028b456f`

## 完整示例

```csharp
using SPTarkov.DI.Annotations;
using SPTarkov.Server.Core.DI;
using SPTarkov.Server.Core.Helpers;
using SPTarkov.Server.Core.Models.Common;
using SPTarkov.Server.Core.Models.Eft.Dialog;
using SPTarkov.Server.Core.Models.Spt.Mod;
using SPTarkov.Server.Core.Models.Utils;
using SPTarkov.Server.Core.Servers;
using SPTarkov.Server.Core.Services;

namespace MyMod;

public record ModMetadata : AbstractModMetadata
{
    public override string ModGuid { get; init; } = "com.example.commands";
    public override string Name { get; init; } = "命令系统示例";
    public override string Author { get; init; } = "开发者";
    public override SemanticVersioning.Version Version { get; init; } = new("1.0.0");
    public override SemanticVersioning.Range SptVersion { get; init; } = new("~4.0.0");
    public override string License { get; init; } = "MIT";
}

// Commando 命令
[Injectable]
public class StatsCommand(
    DatabaseServer databaseServer,
    MailSendService mailSendService
) : IChatCommand
{
    public string CommandPrefix => "stats";
    public List<string> Commands => ["bots", "items"];

    public string GetCommandHelp(string command) => $"用法: stats {command}";

    public ValueTask<string> Handle(string command, UserDialogInfo commandHandler, 
        MongoId sessionId, SendMessageRequest request)
    {
        var tables = databaseServer.GetTables();
        
        var message = command switch
        {
            "bots" => $"机器人类型数量: {tables.Bots.Types.Count}",
            "items" => $"物品总数: {tables.Templates.Items.Count}",
            _ => "未知命令"
        };
        
        mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, message);
        return ValueTask.FromResult(request.DialogId);
    }
}

// SPT 命令
[Injectable]
public class PriceCommand(
    MailSendService mailSendService,
    ItemHelper itemHelper
) : ISptCommand
{
    public string Command => "price";
    public string CommandHelp => "用法: spt price <物品ID> - 查询物品价格";

    public ValueTask<string> PerformAction(UserDialogInfo commandHandler, 
        MongoId sessionId, SendMessageRequest request)
    {
        var parts = request.Text.Split(" ");
        
        if (parts.Length < 3)
        {
            mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, 
                "请提供物品ID");
            return ValueTask.FromResult(request.DialogId);
        }
        
        var itemId = parts[2];
        var item = itemHelper.GetItem(itemId);
        
        if (item.Key && item.Value != null)
        {
            var price = item.Value.Properties?.CreditsPrice ?? 0;
            mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, 
                $"物品: {item.Value.Properties?.Name}\n价格: {price} 卢布");
        }
        else
        {
            mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, 
                $"未找到物品: {itemId}");
        }
        
        return ValueTask.FromResult(request.DialogId);
    }
}
```

## 小贴士

1. **命令要简洁**：前缀和子命令都要短小精悍
2. **提供帮助文本**：让用户知道怎么使用
3. **参数验证**：检查参数数量和格式
4. **友好的错误提示**：命令错误时告诉用户正确用法
5. **日志记录**：记录命令使用情况便于调试
