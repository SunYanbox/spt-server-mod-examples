# 聊天机器人

你可以创建自定义聊天机器人，让玩家在游戏内与它对话。

**源文件：** 
- [20CustomChatBot/](../20CustomChatBot/) - 简单聊天机器人
- [23CustomAbstractChatBot/](../23CustomAbstractChatBot/) - 抽象聊天机器人（支持命令）

## 一、简单聊天机器人

实现 `IDialogueChatBot` 接口创建基础聊天机器人。

### 基本结构

```csharp
using SPTarkov.DI.Annotations;
using SPTarkov.Server.Core.Models.Common;
using SPTarkov.Server.Core.Models.Eft.Dialog;
using SPTarkov.Server.Core.Models.Spt.Mod;
using SPTarkov.Server.Core.Services;

[Injectable]
public class MyChatBot(
    MailSendService mailSendService
) : IDialogueChatBot
{
    // 定义机器人身份
    public UserDialogInfo GetChatBot()
    {
        return new UserDialogInfo
        {
            Id = "my_bot",           // 机器人ID
            Aid = 9999999,            // 账号ID
            Info = new UserDialogDetails
            {
                Nickname = "助手",     // 昵称
                Side = "Usec",        // 阵营
                Level = 69,           // 等级
                MemberCategory = MemberCategory.Sherpa,  // 类型标签
                SelectedMemberCategory = MemberCategory.Sherpa
            }
        };
    }

    // 处理玩家消息
    public ValueTask<string> HandleMessage(MongoId sessionId, SendMessageRequest request)
    {
        // 发送回复
        mailSendService.SendUserMessageToPlayer(
            sessionId,
            GetChatBot(),
            $"你发送了: {request.Text}"
        );

        return ValueTask.FromResult(request.DialogId);
    }
}
```

### MemberCategory 类型

| 类型 | 说明 |
|------|------|
| `Default` | 普通玩家 |
| `Developer` | 开发者（特殊图标） |
| `UniqueId` | 唯一ID |
| `Sherpa` | 向导（特殊图标） |
| `Emissary` | 大使（特殊图标） |

## 二、抽象聊天机器人（支持命令）

继承 `AbstractDialogChatBot` 获得命令处理能力。

### 创建命令类

```csharp
using SPTarkov.DI.Annotations;
using SPTarkov.Server.Core.Models.Common;
using SPTarkov.Server.Core.Models.Eft.Dialog;
using SPTarkov.Server.Core.Services;

// 命令1
[Injectable]
public class HelloCommand(
    MailSendService mailSendService
) : IChatCommand
{
    public string CommandPrefix => "hello";
    public List<string> Commands => ["world"];

    public string GetCommandHelp(string command)
    {
        return "用法: hello world";
    }

    public ValueTask<string> Handle(string command, UserDialogInfo commandHandler, 
        MongoId sessionId, SendMessageRequest request)
    {
        if (command == "world")
        {
            mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, 
                "Hello World! 你好世界！");
        }
        return ValueTask.FromResult(request.DialogId);
    }
}

// 命令2
[Injectable]
public class TimeCommand(
    MailSendService mailSendService
) : IChatCommand
{
    public string CommandPrefix => "time";
    public List<string> Commands => ["now"];

    public string GetCommandHelp(string command)
    {
        return "用法: time now - 显示当前时间";
    }

    public ValueTask<string> Handle(string command, UserDialogInfo commandHandler, 
        MongoId sessionId, SendMessageRequest request)
    {
        if (command == "now")
        {
            mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, 
                $"当前时间: {DateTime.Now:yyyy-MM-dd HH:mm:ss}");
        }
        return ValueTask.FromResult(request.DialogId);
    }
}
```

### 创建机器人

```csharp
using SPTarkov.DI.Annotations;
using SPTarkov.Server.Core.Models.Eft.Dialog;
using SPTarkov.Server.Core.Models.Utils;
using SPTarkov.Server.Core.Services;

[Injectable]
public class MyAdvancedBot(
    ISptLogger<AbstractDialogChatBot> logger,
    MailSendService mailSendService,
    ServerLocalisationService localisationService,
    IEnumerable<IChatCommand> chatCommands  // 自动注入所有命令
) : AbstractDialogChatBot(logger, mailSendService, localisationService, chatCommands)
{
    public override UserDialogInfo GetChatBot()
    {
        return new UserDialogInfo
        {
            Id = "674db14ed849a3727ef24da0",  // 必须是有效的MongoId格式
            Aid = 1234566,
            Info = new UserDialogDetails
            {
                Level = 69,
                MemberCategory = MemberCategory.Developer,
                Nickname = "智能助手",
                Side = "Bear"
            }
        };
    }

    protected override string GetUnrecognizedCommandMessage()
    {
        return "我不明白你在说什么，试试输入: hello world 或 time now";
    }
}
```

## 完整示例

```csharp
using SPTarkov.DI.Annotations;
using SPTarkov.Server.Core.DI;
using SPTarkov.Server.Core.Models.Common;
using SPTarkov.Server.Core.Models.Eft.Dialog;
using SPTarkov.Server.Core.Models.Spt.Mod;
using SPTarkov.Server.Core.Models.Utils;
using SPTarkov.Server.Core.Services;

namespace MyMod;

public record ModMetadata : AbstractModMetadata
{
    public override string ModGuid { get; init; } = "com.example.chatbot";
    public override string Name { get; init; } = "聊天机器人示例";
    public override string Author { get; init; } = "开发者";
    public override SemanticVersioning.Version Version { get; init; } = new("1.0.0");
    public override SemanticVersioning.Range SptVersion { get; init; } = new("~4.0.0");
    public override string License { get; init; } = "MIT";
}

// 简单命令
[Injectable]
public class GreetCommand(
    MailSendService mailSendService
) : IChatCommand
{
    public string CommandPrefix => "greet";
    public List<string> Commands => ["me", "all"];

    public string GetCommandHelp(string command) => $"用法: greet {command}";

    public ValueTask<string> Handle(string command, UserDialogInfo commandHandler, 
        MongoId sessionId, SendMessageRequest request)
    {
        var message = command switch
        {
            "me" => "你好！很高兴见到你！",
            "all" => "大家好！欢迎来到服务器！",
            _ => "未知命令"
        };
        
        mailSendService.SendUserMessageToPlayer(sessionId, commandHandler, message);
        return ValueTask.FromResult(request.DialogId);
    }
}

// 聊天机器人
[Injectable]
public class HelperBot(
    ISptLogger<AbstractDialogChatBot> logger,
    MailSendService mailSendService,
    ServerLocalisationService localisationService,
    IEnumerable<IChatCommand> chatCommands
) : AbstractDialogChatBot(logger, mailSendService, localisationService, chatCommands)
{
    public override UserDialogInfo GetChatBot()
    {
        return new UserDialogInfo
        {
            Id = "674db14ed849a3727ef24da0",
            Aid = 1234566,
            Info = new UserDialogDetails
            {
                Level = 100,
                MemberCategory = MemberCategory.Developer,
                Nickname = "Helper",
                Side = "Bear"
            }
        };
    }

    protected override string GetUnrecognizedCommandMessage()
    {
        return "可用命令: greet me, greet all";
    }
}
```

## 使用方法

1. 在游戏内打开消息界面
2. 找到你的机器人（如"助手"）
3. 发送命令，如 `greet me`

## 小贴士

1. **机器人ID 要用有效的 MongoId**：可以在线生成
2. **命令自动注入**：所有 `IChatCommand` 实现类会被自动发现
3. **友好的错误提示**：在 `GetUnrecognizedCommandMessage` 中告诉用户可用命令
4. **MemberCategory 影响图标**：选择合适的类型让机器人更醒目
